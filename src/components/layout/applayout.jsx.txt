import { useRef } from "react";
import { Outlet, useLocation, useNavigate } from "react-router-dom";
import { LayoutDashboard, Clock, Plus, HandCoins, Briefcase, Settings } from "lucide-react";
import MobileHeader from "./MobileHeader";

const navItems = [
  { path: "/", icon: LayoutDashboard, label: "Home" },
  { path: "/jobs", icon: Briefcase, label: "Jobs" },
  { path: "/add-pay", icon: Plus, label: "Add Pay" },
  { path: "/add-payment", icon: HandCoins, label: "Payment" },
  { path: "/history", icon: Clock, label: "History" },
  { path: "/settings", icon: Settings, label: "Settings" },
];

export default function AppLayout() {
  const location = useLocation();
  const navigate = useNavigate();
  // Remembers the last visited sub-path for each tab root
  const tabStacks = useRef({});

  return (
    <div className="min-h-screen bg-background pb-20 md:pb-0">
      <MobileHeader />
      <main className="max-w-lg mx-auto px-4 pt-20 pb-24">
        <Outlet />
      </main>

      {/* Bottom nav for mobile */}
      <nav
        className="fixed bottom-0 left-0 right-0 bg-card/95 backdrop-blur-lg border-t border-border z-50 block md:hidden"
        style={{ paddingBottom: "env(safe-area-inset-bottom)" }}
      >
        <div className="flex items-center justify-around py-1 max-w-lg mx-auto">
          {navItems.map(({ path, icon: Icon, label }) => {
            const isActive =
              location.pathname === path ||
              (path !== "/" && location.pathname.startsWith(path));

            // Keep stack memory up to date as user navigates
            if (isActive) {
              tabStacks.current[path] = location.pathname;
            }

            return (
              <button
                key={path}
                onClick={() => {
                  if (isActive) {
                    // Tapping active tab resets to root and scrolls up
                    tabStacks.current[path] = path;
                    navigate(path);
                    window.scrollTo({ top: 0, behavior: "smooth" });
                  } else {
                    // Navigate to last remembered path in this tab (stack preservation)
                    navigate(tabStacks.current[path] || path);
                  }
                }}
                style={{ userSelect: "none" }}
                className={`flex flex-col items-center gap-0.5 px-2 py-1.5 rounded-xl transition-colors min-w-[44px] min-h-[44px] justify-center ${
                  isActive ? "text-primary" : "text-muted-foreground hover:text-foreground"
                }`}
              >
                <Icon className="w-5 h-5" />
                <span className="text-[10px] font-medium">{label}</span>
              </button>
            );
          })}
        </div>
      </nav>
    </div>
  );
}
